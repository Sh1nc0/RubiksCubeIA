functor (S : STATE) (D : OPERATION) (C : Additive.ADDITIVE) ->
  sig
    type state = S.state
    type op = D.op
    type num = C.c
    type transition = {
      startState : state;
      operation : op;
      endState : state;
      cost : num;
    }
    type successors = state -> transition list
    type state_problem = {
      successors : successors;
      isFinalState : state -> bool;
    }
    type extension_state_problem = {
      states : state list;
      theInitialState : state;
      finalStates : state list;
      edges : (state * op * state * num) list;
    }
    val mkExtensionStateProblem :
      ss:state list ->
      i:state ->
      fs:state list ->
      es:(state * op * state * num) list -> extension_state_problem
    val isExtensionStateProblem : extension_state_problem -> bool
    type solution = transition list
    val isSolution :
      t:(state -> transition list) ->
      f:(state -> bool) -> i:state -> ts:solution -> bool
    val solutionCost : solution -> num
    val childrenStates :
      p:extension_state_problem -> state -> transition list
    val isAFinalState : extension_state_problem -> state -> bool
    val build :
      n:int ->
      t:successors ->
      f:(state -> bool) ->
      i:state ->
      ss:state list ->
      os:(state * op * state * num) list ->
      q:state list * state list ->
      state list * (state * op * state * num) list
    val extension2comprehension : extension_state_problem -> state_problem
    val comprehension2extension :
      n:int ->
      t:(state -> transition list) ->
      f:(state -> bool) -> i:state -> extension_state_problem
    val string_of_state : state -> string
    val string_of_op : op -> string
    val string_of_transition : transition -> string
    val string_of_solution : solution -> string
    val short_solution : solution -> string
  end
